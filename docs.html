()put — چاپ و خروجی

قدرتمندترین تابع چاپ در تاریخ برنامه‌نویسی فارسی!

# مثال کامل تمام قابلیت‌های put()
str name = ^Ali^
int age = 21
arr scores = i{95, 88, 92}

put(^سلام ^, upper(name), ^! شما ^, age, ^ سالته دارید^)
# → سلام ALI! شما 21 سالته دارید

put(f^نام: {name} | میانگین نمرات: {~sum(scores)/len(scores)~}^)
# → نام: Ali | میانگین نمرات: 91.666...

put(len(^دیم بهترین زبانه^), ^ کاراکتر^)           # → 15 کاراکتر
put(name=>[0:1], ^ تا ^, name=>[1:])                 # → A تا li
put(~3.14 * 5 * 5~)                                 # → 78.5
f-string len() upper()/lower() =>[slice] ~eval~

()inp — ورودی هوشمند

# تمام حالت‌های inp در یک مثال
inp(name, ^اسم شما: ^)                    # رشته معمولی
inp(age, i^سن شما (عدد): ^)                # فقط عدد صحیح
inp(height, d^قد شما (متر): ^)             # عدد اعشاری
inp(is_student, b^دانشجو هستید؟ (yes/no): ^)  # بولین هوشمند
inp(password, f^رمز برای {name}: ^)        # پیام فرمتغیر

# حتی بدون متغیر (فقط نمایش پیام)
inp(¿, ^برای ادامه Enter بزنید...^)
i^ d^ b^ f^ پیام فرمت‌شده تشخیص خودکار yes/no

()time — زمان و تایمر

time()                    # یونیکس تایم
time(now)                 # تاریخ و ساعت فعلی
time(clock)               # ساعت دیجیتال زنده
time(countdown,10)        # شمارش معکوس ۱۰ ثانیه
time(timer)               # شروع/ن/توقف تایمر
time(delay,2000)          # مکث ۲ ثانیه
