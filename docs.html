()put — چاپ و خروجی قدرتمند

پیشرفته‌ترین تابع چاپ با پشتیبانی از f-string، محاسبات، برش رشته، توابع متنی و ... 
# تمام قابلیت‌های put() در یک نگاه
str name = ^Ali Reza^
int age = 21
arr scores = i{95, 88, 92}

put(^سلام ^, upper(name), ^! شما ^, age, ^ سالته دارید^)
# → سلام ALI REZA! شما 21 سالته دارید

put(f^نام: {name} | میانگین: {~sum(scores)/len(scores)~}^)
# → نام: Ali Reza | میانگین: 91.666...

put(len(^دیم بهترین زبان است^), ^ کاراکتر^)           # → 18 کاراکتر
put(name=>[0:3], ^ تا ^, name=>[4:])                   # → Ali تا Reza
put(lower(^DIM^))                                      # → dim
put(~3.14 * 10 * 10~)                                  # → 314.0
f-string len() upper()/lower() =>[slice] ~eval~

()inp — ورودی هوشمند

ورودی با تشخیص خودکار نوع داده، پیام فرمت‌شده و پشتیبانی از بولین فارسی 
# تمام حالت‌های inp()
inp(name, ^اسم شما چیست؟ ^)                    # رشته
inp(age, i^سن شما (عدد): ^)                     # عدد صحیح
inp(height, d^قد شما (متر): ^)                  # عدد اعشاری
inp(active, b^آیا فعال هستید؟ (yes/no): ^)     # بولین هوشمند
inp(pass, f^رمز عبور برای {name}: ^)           # پیام با متغیر

inp(¿, ^برای ادامه Enter بزنید...^)             # فقط نمایش پیام
i^ d^ b^ f^ فرمت پیام yes/no خودکار

str = — تعریف رشته

تعریف رشته با پشتیبانی از الحاق، برش، f-string و متن چندخطی 
str greeting = ^سلام دنیا^
str part = greeting=>[6:]                 # → دنیا
str full = ^سلام ^ + name + ^ عزیز^
str today = f^امروز {~time(now)~} است^

str poem = """
دیم زبان آینده است
زیبا، ساده و قدرتمند
ساخت ایران
"""

int = • dou = • bol =

تعریف متغیرهای عددی و بولین با محاسبه مستقیم 
int year = 1404
dou pi = 3.1415926535
bol is_active = true
bol adult = ~age >= 18~          # محاسبه بولین
bol from_str = ^yes^             # تبدیل خودکار

arr = — آرایه‌ها

سه نوع آرایه: عددی، رشته‌ای و محاسباتی 
arr nums   = i{10, 20, 30, 40}
arr names  = s{^Ali^, ^Reza^, "Sara"}
arr calc   = m{1+2, age*2, nums[0]}

put(names[1])        # → Reza
put(names[^Ali^])    # → 0 (جستجو)

con$ — ثابت‌ها

مقادیر ثابت و غیرقابل تغییر در طول برنامه 
con$pi = 3.14159
con$app_name = ^دیم^
con$version = 1.4
con$author = ^محمد ابوالفضل^

# خطا: con$pi = 3 → غیرمجاز!

IF • ELSEIF • ELSE • LOOP

کنترل جریان با سینتکس ساده و قدرتمند 
IF(age >= 18)>>{
    put(^شما بالغ هستید^)
}ELSEIF(age >= 13)>>{
    put(^نوجوان^)
}ELSE{
    put(^کودک^)
}

int count = 0
LOOP(count < 5)>>{
    put(^شمارش: ^, count)
    int count = ~count + 1~
}

()time — زمان و تایمر

time()                    # یونیکس تایم

time(timer)               # تایمر دستی
time(delay,1500)          # مکث 1.5 ثانیه
